from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup, WebAppInfo from telegram.ext import Updater, CommandHandler, CallbackContext def start(update: Update, context: CallbackContext) -> None: web_app = WebAppInfo(url="https://your-web-app-url.com") # Ссылка на твой веб приложение button = InlineKeyboardButton(text="Открыть mini app", web_app=web_app) keyboard = InlineKeyboardMarkup([[button]]) update.message.reply_text("Нажмите на кнопку ниже, чтобы открыть мини-приложение:", reply_markup=keyboard) def main(): token = "YOUR_BOT_TOKEN" updater = Updater(token) dispatcher = updater.dispatcher dispatcher.add_handler(CommandHandler("start", start)) updater.start_polling() updater.idle() if __name__ == "__main__": main()
